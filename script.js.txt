let playerMini={level:1,exp:0,bodyType:'normal',expToNext:100};
const weightInput=document.getElementById('weight');
const unitSelect=document.getElementById('weightUnit');
let currentUnit='jin';
const exerciseInput=document.getElementById('exerciseCalories');
const targetInput=document.getElementById('targetCalories');
const proteinEl=document.getElementById('protein');
const carbEl=document.getElementById('carb');
const fatEl=document.getElementById('fat');
const caloriesEl=document.getElementById('calories');
const expEl=document.getElementById('exp');
const levelEl=document.getElementById('level');
const goalDateEl=document.getElementById('goalDate');
const miniImg=document.getElementById('mini');
const weightChart=document.getElementById('weightChart');
const ctx=weightChart.getContext('2d');

let records=JSON.parse(localStorage.getItem('records')||'[]');
let meals=[];

// 单位切换
unitSelect.addEventListener('change',()=>{
  const oldUnit=currentUnit;
  currentUnit=unitSelect.value;
  let val=parseFloat(weightInput.value)||0;
  if(oldUnit==='jin' && currentUnit==='kg') weightInput.value=(val/2).toFixed(1);
  else if(oldUnit==='kg' && currentUnit==='jin') weightInput.value=(val*2).toFixed(1);
  updateChart(); predictGoal();
});

// 小人动画
function updateMini(){
  miniImg.src=playerMini.exp>200
    ?'https://user3695.cn.imgto.link/public/20260306/mini-slim-png.avif'
    :'https://user3695.cn.imgto.link/public/20260306/mini-normal-png.avif';
  expEl.textContent=Math.floor(playerMini.exp);
  levelEl.textContent=playerMini.level;
}

// 小人动作
function jumpMini(){ miniImg.src='https://user3695.cn.imgto.link/public/20260306/mini-jump-png.avif'; miniImg.style.transform='translateY(-20px)'; setTimeout(()=>{ miniImg.src='https://user3695.cn.imgto.link/public/20260306/mini-normal-png.avif'; miniImg.style.transform='translateY(0)'; },500);}
function eatMini(){ miniImg.src='https://user3695.cn.imgto.link/public/20260306/mini-eat-png.avif'; setTimeout(()=>{ miniImg.src='https://user3695.cn.imgto.link/public/20260306/mini-normal-png.avif'; },500);}
function randomMiniAction(){
  const actions=[jumpMini,eatMini,()=>{miniImg.style.transform='translateY(0) rotate(5deg)';setTimeout(()=>miniImg.style.transform='translateY(0) rotate(0deg)',300);},()=>{miniImg.style.transform='scale(1.05)';setTimeout(()=>miniImg.style.transform='scale(1)',300);}];
  const act=actions[Math.floor(Math.random()*actions.length)]; act();
}
miniImg.addEventListener('click', randomMiniAction);

// 饮食记录
function calculateMealCalories(){ let total=0; meals.forEach(meal=>{total+=meal.gram*meal.caloriesPerGram;}); document.getElementById('mealCalories').textContent=total.toFixed(0); return total;}
function addMeal(name,gram,calPerGram){meals.push({name,gram,caloriesPerGram:calPerGram}); calculateMealCalories(); autoSave();}
document.querySelectorAll('.addMealBtn').forEach(btn=>{
  btn.addEventListener('click',()=>{
    const parent=btn.parentElement;
    const nameEl=parent.querySelector('.mealName');
    const gramEl=parent.querySelector('.mealGram');
    const name=nameEl.value;
    const gram=parseFloat(gramEl.value)||0;
    const kcalPerGram=2;
    addMeal(name,gram,kcalPerGram);
    nameEl.value=''; gramEl.value='';
  });
});

// 总热量计算
function updateCalories(){
  const foodCalories=calculateMealCalories();
  const exerciseCalories=parseFloat(exerciseInput.value)||0;
  const totalCalories=foodCalories-exerciseCalories;
  caloriesEl.textContent=Math.max(0,totalCalories.toFixed(0));
  proteinEl.textContent=Math.round(foodCalories*0.3/4);
  carbEl.textContent=Math.round(foodCalories*0.5/4);
  fatEl.textContent=Math.round(foodCalories*0.2/9);
  return totalCalories;
}

// 保存记录
function autoSave(){
  const today=new Date().toLocaleDateString();
  const record={date:today,weight:parseFloat(weightInput.value)||0,meals:meals,exerciseCalories:parseFloat(exerciseInput.value)||0};
  const idx=records.findIndex(r=>r.date===today);
  if(idx>=0) records[idx]=record; else records.push(record);
  localStorage.setItem('records',JSON.stringify(records));
  playerMini.exp+=10;
  if(playerMini.exp>=playerMini.expToNext){playerMini.level++; playerMini.exp-=playerMini.expToNext; playerMini.expToNext=playerMini.level*100;}
  updateMini(); updateChart(); predictGoal(); randomMiniAction();
}

// 体重曲线
function getWeightForChart(w){return currentUnit==='jin'?w:w/2;}
function updateChart(){ctx.clearRect(0,0,weightChart.width,weightChart.height);if(records.length<2)return;ctx.beginPath();ctx.strokeStyle='#ff6f91'; let stepX=weightChart.width/(records.length-1); let max=Math.max(...records.map(r=>getWeightForChart(r.weight))); let min=Math.min(...records.map(r=>getWeightForChart(r.weight))); records.forEach((r,i)=>{let y=weightChart.height-(getWeightForChart(r.weight)-min)/(max-min)*weightChart.height;if(i===0)ctx.moveTo(0,y);else ctx.lineTo(i*stepX,y);}); ctx.stroke();}

// 预测目标体重
function predictGoal(){
  if(records.length<2)return;
  const netCal=updateCalories();
  const lastWeight=getWeightForChart(records[records.length-1].weight);
  const targetWeight=currentUnit==='jin'?90:45;
  const dailyWeightChange=netCal/7700*2;
  if(dailyWeightChange<=0){goalDateEl.textContent='-'; return;}
  const daysNeeded=(lastWeight-targetWeight)/dailyWeightChange;
  let goalDate=new Date(); goalDate.setDate(goalDate.getDate()+daysNeeded);
  goalDateEl.textContent=goalDate.toLocaleDateString();
}

// 按钮事件
[weightInput,exerciseInput].forEach(inp=>inp.addEventListener('input',autoSave));
document.getElementById('jumpBtn').addEventListener('click',jumpMini);
document.getElementById('eatBtn').addEventListener('click',eatMini);

// 初始化
updateMini(); updateChart(); predictGoal();
